type Vector is array[50] real;
type Matrix is array[150] Vector;
type SDFDF is real;

var as: SDFDF;
var A: Matrix;
var B: Matrix; 
var C: Matrix;
var result: Vector;
var temp: real;

routine matrixMultiply() is
    var i: integer;
    var j: integer; 
    var k: integer;
    as := 0;
    for i in 1..150 loop
        for j in 1..50 loop
            temp := 0.0;
            for k in 1..150 loop
                temp := temp + A[i][k] * B[k][j];
            end
            C[i][j] := temp;
        end
    end
    
    A[1][1] := 1.0;
    A[99][99] := 1.0;
    B[1][99] := 2.0;
    C[50][50] := 3.0;
end

routine dangerousAccess() is
    A[100][0] := 5.0;
    B[-1][50] := 10.0;
    C[0][150] := 7.0;
end
