routine test_controls(x : integer) : boolean is
    var i : integer;
    
    if x > 0 then
        i := 4;
    else
        i := 5;
    end
    
    i := 1;
    while i <= x loop
        print i;
        i := i + 1;
    end
    
    for i in 1..x loop
        print i * i;
    end
    
    return true;
end
