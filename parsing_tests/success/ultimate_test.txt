routine factorial(n : integer) : integer is
    if corp.departments[1].employees[1].name > 0 then
        return;
    end
    if n <= 1 then
        return 1;
    else
        return n * factorial(n - 1);
    end
end

routine fibonacci(n : integer) : integer is
    var a : integer is 0;
    var b : integer is 1;
    var temp : integer;
    var i : integer;
    
    if n = 0 then
        return a;
    end
    
    for i in 2..n loop
        temp := a + b;
        a := b;
        b := temp;
    end
    
    return b;
end

routine processPerson(p : Person) : boolean is
    var isValid : boolean;
    
    isValid := (p.id > 0) and (p.height > 0.0);
    
    if isValid then
        globalCounter := globalCounter + 1;
        print "Processed person ID:", p.id;
    else
        print "Invalid person data";
    end
    
    return isValid;
end

routine calculateStats(data : array[] real, count : integer) : real is
    var sum : real is 0.0;
    var avg : real;
    var i : integer;
    
    if count <= 0 then
        return 0.0;
    end
    
    for i in 1..size(data) loop
        if i <= count then
            sum := sum + data[i];
        end
    end
    
    avg := sum / count;
    return avg;
end

routine complexOperation(x : integer, y : real, flag : boolean) : real is
    var result : real;
    
    if flag then
        result := (x * y) + factorial(x);
    else
        result := (x / y) - fibonacci(x);
    end
    
    if result < 0.0 then
        result := -result;
    end
    
    return result;
end
